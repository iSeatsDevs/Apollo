import { action } from '@storybook/addon-actions';
import { Meta, Story, Props, Preview } from '@storybook/addon-docs/blocks';
import { Calendar } from '../index';

<Meta title="Components/Calendar" component={Calendar} />

<Props of={Calendar} />

## Default

<Preview>
  <Story name="Default">
    <Calendar onChange={action('Calenda changed')} />
  </Story>
</Preview>

## Default value

<Preview>
  <Story name="Default value">
    <Calendar onChange={action('Calenda changed')} defaultValue={new Date('3005-01-01')} defaultOpenDate={new Date('3005-01-01')} />
  </Story>
</Preview>

## Year

If you set the `showYear` prop, the year will be displayed as an input field.

<Preview>
  <Story name="Year">
    <Calendar onChange={action('Calenda changed')} showYear />
  </Story>
</Preview>

## Default open date

You can change the default date the calendar will be opened to by setting
the `defaultOpenDate` prop.

<Preview>
  <Story name="Default open date">
    <Calendar onChange={action('Calenda changed')} defaultOpenDate={new Date('3005-01-01')} showYear/>
  </Story>
</Preview>

## Min

Any date before the date assigned will be disabled.

<Preview>
  <Story name="Min">
    <Calendar onChange={action('Calenda changed')} min={new Date} />
  </Story>
</Preview>

## Max

Any date after the date assigned will be disabled.

<Preview>
  <Story name="Max">
    <Calendar onChange={action('Calenda changed')} max={new Date} />
  </Story>
</Preview>

## Only

The `only` prop is used for selecting a range of dates that are only accessible, 
the dates not assigned will be disabled.

<Preview>
  <Story name="Only">
    <Calendar onChange={action('Calenda changed')} only={[new Date('2020-01-20'), new Date('2020-01-21'), new Date('2020-01-22')]}/>
  </Story>
</Preview>
